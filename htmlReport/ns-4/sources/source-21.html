 /**

  * no more specific type can be thrown.
  * 
  * @since 2.1

 		super(message);
 	}
 	
 	public OrderEntryException(String message, Throwable cause) {
 		super(message, cause);
 	}
 	
 	public OrderEntryException(String messageKey, Object[] parameters) {
 		super(messageKey, parameters);
 	}
 	
 }
